library(ggplot2)

sigmoid <- function(x){1 / (1 + exp(-x))}

compute_asymptotic_map <- function(betahat, sehat, prior_variance){
  tau <- 1/(sehat^2)
  tau0 <- 1/prior_variance
  mu <- tau/(tau + tau0) * betahat
  var <- 1/(tau + tau0)
  return(list(mu=mu, var=var))
}

compute_asymptotic_map2 <- function(fit, prior_variance){
  res <- unname(summary(fit)$coef[2,])
  compute_asymptotic_map(res[1], res[2], prior_variance)
}

sim <- function(b0 = -2, b = 1, n = 50){
  # simulate
  x <- rnorm(n)
  logits <- b0 + x * b
  probs <- sigmoid(logits)
  y <- rbinom(length(probs), 1, probs)
  
  # prior variance grid
  sigma2s <- 2^seq(-8, 6, by=0.1) # prior variances
  lambdas <- 1/(length(y) * sigma2s) # equivalent l2 penalties
  
  # asymptotic approximation
  fit.mle <- glm(y ~ x, family='binomial')
  
  # MAP, fix intercept to MLE
  fit.glmnet <- glmnet::glmnet(cbind(x, x), y, 
                               standardize = F, 
                               intercept=F,
                               offset = rep(1, length(y)) * fit.mle$coefficients[1],
                               family='binomial', 
                               alpha=0, 
                               lambda=lambdas)
  
  
  res <- purrr::map_dfr(sigma2s, ~compute_asymptotic_map2(fit.mle, .x)) %>%
    mutate(
      map = unname(Matrix::colSums(coef(fit.glmnet))),
      sigma2 = sigma2s,
      n = n,
      b0 = b0,
      b = b
    )
  return(res)
}

sim2 <- function(pi0=0.1, b0 = -2, b = 1, n = 500, sigma2_min=0.0001, sigma2_max=10){
  # simulate
  x <-rbinom(n, 1, 1-pi0)
  print(mean(x))
  logits <- b0 + x * b
  probs <- sigmoid(logits)
  y <- rbinom(length(probs), 1, probs)
  
  # prior variance grid
  sigma2s <- 10^seq(log10(sigma2_min), log10(sigma2_max), length.out=100) # prior variances
  lambdas <- 1/(n * sigma2s) # equivalent l2 penalties
  
  # asymptotic approximation
  fit.mle <- glm(y ~ x, family='binomial')
  
  # MAP, fix intercept to MLE
  fit.glmnet <- glmnet::glmnet(cbind(x, x), y, 
                               standardize = F, 
                               intercept=F,
                               offset = rep(1, length(y)) * fit.mle$coefficients[1],
                               family='binomial', 
                               alpha=0, 
                               lambda=lambdas)
  
  
  res <- purrr::map_dfr(sigma2s, ~compute_asymptotic_map2(fit.mle, .x)) %>%
    mutate(
      map = unname(Matrix::colSums(coef(fit.glmnet))),
      sigma2 = sigma2s,
      n = n,
      b0 = b0,
      b = b
    )
  return(res)
}
